  5 /**
  6  * @class   Service class that will retrieve AtProteome data for this entry given an AGI.
  7  *          To retrieve data from AtProteome, two requests need to be made to the remote
  8  *          server. The first request ascertains the internal ID used by AtProteome for
  9  *          the given AGI. A second request then retrieves the data from AtProteome.
 10  *          Data is received in XML format, and will need to go through a tidying proxy
 11  *          if retrieving data directly from AtProteome.
 12  * @description Default class constructor
 13  * @param   {String} agi            Agi to look up
 14  * @param   {String} endpointURL    Endpoint URL for this service
 15  * @extends MASCP.Service
 16  */
 17 MASCP.AtProteomeReaderJson = MASCP.buildService(function(data) {
 18                         this._raw_data = data;
 19                         if (data) {
 20                             this._populate_spectra(data);
 21                             this._populate_peptides(data);
 22                         }
 23                         return this;
 24                     });
 25 
 26 MASCP.AtProteomeReaderJson.prototype.requestData = function()
 27 {
 28     var self = this;
 29     var agi = this.agi;
 30     return {
 31         type: "POST",
 32         dataType: "json",
 33         data: { 'agi'       : agi,
 34                 'service'   : 'atproteome-json' 
 35         }
 36     };
 37 };

 56 MASCP.AtProteomeReaderJson.Result.prototype = jQuery.extend(MASCP.AtProteomeReaderJson.Result.prototype,
 57 /** @lends MASCP.AtProteomeReaderJson.Result.prototype */
 58 {
 59     /** @field 
 60      *  @description Hash keyed by tissue name containing the number of spectra for each tissue for this AGI */
 61     spectra :   null,
 62     /** @field
 63      *  @description Hash keyed by tissue name containing the number of spectra for each peptide (keyed by "start-end" position) */
 64     peptide_counts_by_tissue : null
 65 });
 66 
 67 MASCP.AtProteomeReaderJson.Result.prototype._populate_spectra = function(data)
 68 {
 69     this.spectra = {};
 70     if ( ! data || ! data.tissues ) {
 71         return;
 72     }
 73     for (var i = 0; i < data.tissues.length; i++ ) {
 74         this.spectra[data.tissues[i].sequence] = parseInt(data.tissues[i].qty_spectra);
 75     }
 76 };
 77 
 78 MASCP.AtProteomeReaderJson.Result.prototype._populate_peptides = function(data)
 79 {
 80     this.peptide_counts_by_tissue = {};
 81     if ( ! data || ! data.peptides ) {
 82         return;
 83     }
 84     for (var i = 0; i < data.peptides.length; i++ ) {
 85         var a_peptide = data.peptides[i];
 86         var peptide_position = a_peptide.position+'-'+(parseInt(a_peptide.position)+parseInt(a_peptide.sequence.length));
 87         for (var j = 0; j < a_peptide.tissues.length; j++ ) {
 88             var a_tissue = a_peptide.tissues[j];
 89             if (! this.peptide_counts_by_tissue[a_tissue.tissue]) {
 90                 this.peptide_counts_by_tissue[a_tissue.tissue] = {};
 91             }
 92             this.peptide_counts_by_tissue[a_tissue.tissue][peptide_position] = parseInt(a_tissue.qty_spectra);
 93         }
 94     }
 95 };
