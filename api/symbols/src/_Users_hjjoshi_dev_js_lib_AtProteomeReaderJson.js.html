  5 /**
  6  * @class   Service class that will retrieve AtProteome data for this entry given an AGI.
  7  *          To retrieve data from AtProteome, two requests need to be made to the remote
  8  *          server. The first request ascertains the internal ID used by AtProteome for
  9  *          the given AGI. A second request then retrieves the data from AtProteome.
 10  *          Data is received in XML format, and will need to go through a tidying proxy
 11  *          if retrieving data directly from AtProteome.
 12  * @description Default class constructor
 13  * @param   {String} agi            Agi to look up
 14  * @param   {String} endpointURL    Endpoint URL for this service
 15  * @extends MASCP.Service
 16  */
 17 MASCP.AtProteomeReaderJson = MASCP.buildService(function(data) {
 18                         this._raw_data = data;
 19                         if (data) {
 20                             this._populate_spectra(data);
 21                             this._populate_peptides(data);
 22                         }
 23                         return this;
 24                     });

 26 MASCP.AtProteomeReaderJson.prototype.requestData = function()
 27 {
 28     var self = this;
 29     var agi = this.agi;
 30     return {
 31         type: "POST",
 32         dataType: "json",
 33         data: { 'agi'       : agi,
 34                 'service'   : 'atproteome-json' 
 35         }
 36     };
 37 };

 58 MASCP.AtProteomeReaderJson.Result.prototype = jQuery.extend(MASCP.AtProteomeReaderJson.Result.prototype,
 59 /** @lends MASCP.AtProteomeReaderJson.Result.prototype */
 60 {
 61     /** @field 
 62      *  @description Hash keyed by tissue name containing the number of spectra for each tissue for this AGI */
 63     spectra :   null,
 64     /** @field
 65      *  @description Hash keyed by tissue name containing the number of spectra for each peptide (keyed by "start-end" position) */
 66     peptide_counts_by_tissue : null,
 67     /** @field
 68      *  @description String containing the sequence for the retrieved AGI */
 69     sequence : null

 72 MASCP.AtProteomeReaderJson.Result.prototype._populate_spectra = function(data)
 73 {
 74     this.spectra = {};
 75     this._tissues = [];
 76     if ( ! data || ! data.tissues ) {
 77         return;
 78     }
 79     for (var i = 0; i < data.tissues.length; i++ ) {
 80         this._tissues[i] = data.tissues[i].tissue;
 81         this.spectra[data.tissues[i].tissue] = parseInt(data.tissues[i].qty_spectra);
 82     }
 83 };
 84 
 85 MASCP.AtProteomeReaderJson.Result.prototype._populate_peptides = function(data)
 86 {
 87     this.peptide_counts_by_tissue = {};
 88     if ( ! data || ! data.peptides ) {
 89         return;
 90     }
 91         
 92     this.sequence = data.sequence;
 93     
 94     for (var i = 0; i < data.peptides.length; i++ ) {
 95         var a_peptide = data.peptides[i];
 96         var peptide_position = a_peptide.position+'-'+(parseInt(a_peptide.position)+parseInt(a_peptide.sequence.length));
 97         for (var j = 0; j < a_peptide.tissues.length; j++ ) {
 98             var a_tissue = a_peptide.tissues[j];
 99             if (! this.peptide_counts_by_tissue[a_tissue.tissue]) {
100                 this.peptide_counts_by_tissue[a_tissue.tissue] = {};
101             }
102             this.peptide_counts_by_tissue[a_tissue.tissue][peptide_position] = parseInt(a_tissue.qty_spectra);
103         }
104     }
105 };

121 MASCP.AtProteomeReaderJson.prototype._rendererRunner = function(sequenceRenderer) {
122     var tissues = this.result? this.result.tissues() : [];
123     for (var tiss in tissues) {
124         var tissue = tissues[tiss];
125         if (this.result.spectra[tissue] < 1) {
126             continue;
127         }
128         var peptide_counts = this.result.peptide_counts_by_tissue[tissue];
129         var simple_tissue = tissue.replace(/\s/g,'');
130         var overlay_name = 'atproteome_by_tissue_'+simple_tissue;
131     	
132         // var css_block = ' .overlay { display: none; } .active .overlay { display: block; top: 0px; background: #000099; } ';
133         
134     	var css_block = ' .overlay { display: none; } .tracks .active { fill: #000099; } .inactive { display: none; } .active .overlay { display: block; top: 0px; background: none; border-bottom: solid #000000 1px; } ';
135     	
136     	MASCP.registerLayer(overlay_name,{ 'fullname' : tissue + ' ('+this.result.spectra[tissue]+' spectra)', 'group' : 'atproteome', 'color' : '#000099', 'css' : css_block });
137 
138         var do_diagrams = (window.location.search.replace(/^\?/, '').indexOf('drawMap') >= 0);
139 
140         if (typeof GOMap != 'undefined' && do_diagrams) {
141             var map = this._map;
142             if ( ! map ) {
143                 map = new GOMap.Diagram('mature_flower_diagram.svg');
144                 var map_container = jQuery('<div style="position: relative; height: 0px; width: 100%; margin-bottom: 2px; overflow: hidden;"></div>');
145                 map_container.bind('load', function() {
146                     map_container.css({'height': '100%','overflow':'visible'});
147                 });
148                 this._map = map;
149                 this._map_container = map_container[0];
150                 map.appendTo(map_container[0]);
151                 
152             }
153             
154             // FIXME FOR MULTIPLE BINDINGS
155             
156             jQuery(MASCP.getLayer(overlay_name)).bind('mouseover',function() {
157                 map.showKeyword(simple_tissue);
158             });
159         }
160         
161         
162     	var positions = this._normalise(this._mergeCounts(peptide_counts));
163     	var index = 0;
164     	var last_start = null;
165     	while (index <= positions.length) {
166     	    if ((! (positions[index] > 0) || (index == positions.length) ) && last_start != null) {
167     	        sequenceRenderer.getAminoAcidsByPosition([last_start])[0].addBoxOverlay(overlay_name,1,index-1-last_start);
168     	        last_start = null;
169     	    }
170     	    if (positions[index] > 0 && last_start == null) {
171     	        last_start = index;
172     	    }
173     	    index += 1;
174     	}
175     }
176 };

178 MASCP.AtProteomeReaderJson.prototype.setupSequenceRenderer = function(sequenceRenderer)
179 {
180 	MASCP.registerGroup('atproteome',{ 'fullname' : 'AtProteome data','hide_member_controllers' : true, 'hide_group_controller' : true, 'color' : '#000099' });
181 
182     var reader = this;
183 
184     this.bind('resultReceived', function() {
185         if ( sequenceRenderer.sequence != this.result.sequence ) {
186             jQuery(sequenceRenderer).bind('sequenceChange',function() {
187                 jQuery(sequenceRenderer).unbind('sequenceChange',arguments.callee);
188                 reader._rendererRunner(sequenceRenderer);
189                 jQuery(sequenceRenderer).trigger('resultsRendered',[reader]);
190             });
191             sequenceRenderer.setSequence(this.result.sequence);
192         } else {
193             reader._rendererRunner(sequenceRenderer);
194             jQuery(sequenceRenderer).trigger('resultsRendered',[reader]);            
195         }
196     });
197 
198     return this;
199 };

201 MASCP.AtProteomeReaderJson.prototype._normalise = function(array)
202 {
203     var max_val = 0;
204     for (var i = 0; i < array.length; i++)
205     {
206         if (array[i] && array[i] > max_val) {
207             max_val = array[i];
208         }
209     }
210     for (var i = 0; i < array.length; i++)
211     {
212         if (array[i] && array[i] > 0) {
213             array[i] = (array[i] * 1.0) / max_val;
214         }
215     }
216     return array;
217 };

219 MASCP.AtProteomeReaderJson.prototype._mergeCounts = function(hash)
220 {
221     var counts = [];
222     for (var position in hash) {
223         var ends = position.split('-');
224         var start = parseInt(ends[0]);
225         var end = parseInt(ends[1]);
226         for (var i = start; i <= end; i++) {
227             if ( ! counts[i] ) {
228                 counts[i] = 0;
229             }
230             counts[i] += hash[position];
231         }
232     }
233     return counts;
234 };
