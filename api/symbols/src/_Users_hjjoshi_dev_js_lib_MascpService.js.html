 34 /** Build a data retrieval class that uses the given function to extract result data.
 35  *  @static
 36  *  @param  {Function}  dataExtractor   Function to extract data from the resultant data (passed as an argument
 37  *                                      to the function), and then populate the result object. The function is
 38  *                                      bound to a hash to populate data in to. When no data is passed to the
 39  *                                      function, the hash should be populated with default values.
 40  */
 41 MASCP.buildService = function(dataExtractor)
 42 {
 43     var clazz = function(agi,endpointURL)
 44     {
 45         if (typeof endpointURL != 'undefined') {
 46             this._endpointURL = endpointURL;
 47         } else {
 48             this._endpointURL = clazz.SERVICE_URL;
 49         }
 50         this.agi = agi;
 51         return this;
 52     };
 53 
 54     clazz.Result = function(data)
 55     {
 56         jQuery.extend(this,dataExtractor.apply(this,[data]));
 57     };
 58     
 59     clazz.prototype = jQuery.extend(new MASCP.Service(),{
 60         '__class__'       :       clazz,
 61         '__result_class'  :       clazz.Result,
 62         '_endpointURL'    :       null
 63     });
 64     
 65     clazz.Result.prototype = jQuery.extend(new MASCP.Service.Result(),{
 66        '__class__'        :       clazz.Result
 67     });
 68 
 69     clazz.Result.prototype = jQuery.extend(clazz.Result.prototype,dataExtractor.apply({},[]));
 70     return clazz;
 71 };

218 MASCP.Service.prototype._dataReceived = function(data,status)
219 {
220     var clazz = this.__result_class;
221 
222     if ( ! this.result ) {
223         result = new clazz(data);
224         this.result = result;
225     } else {
226         var new_result = new clazz(data);
227         jQuery.extend( this.result, new_result );        
228     }
229     this.result.reader = this;
230     this.result.agi = this.agi;
231 };

275 /**
276  *  Asynchronously retrieves data from the remote source. When data is received, a 
277  *  resultReceived.mascp event is triggered upon this service, while an error.mascp
278  *  event is triggered when an error occurs. This method returns a reference to self
279  *  so it can be chained.
280  */
281 MASCP.Service.prototype.retrieve = function()
282 {
283     var self = this;
284 
285     var request_data = this.requestData();
286     if (! request_data ) {
287         return this;
288     }
289     request_data = jQuery.extend({
290     async:      this.async,
291     url:        this._endpointURL,
292     error:      function(response,req,settings) {
293                     jQuery(self).trigger("error");
294                     throw "Error occurred retrieving data for service "+self._endpointURL;
295                 },
296     success:    function(data,status) {
297                     self._dataReceived(data,status);
298                     jQuery(self).trigger("resultReceived");
299                     jQuery(MASCP.Service).trigger("resultReceived");
300                 }
301     },request_data);
302     
303     if (jQuery.browser.msie && window.XDomainRequest) {
304         this._retrieveIE(request_data);
305         return this;
306     }
307     
308     jQuery.ajax(request_data);
309     return this;
310 };

312 /**
313  * Private method for performing a cross-domain request using Internet Explorer 8 and up. Adapts the 
314  * parameters passed to the jQuery method, and builds an XDR object. There is no support for a locking
315  * synchronous method to do these requests (that is required for Unit testing) so an alert box is used
316  * to provide the locking.
317  * @private
318  * @param {Object} dataHash Hash with the data and settings used to build the query.
319  */
320 
321 
322 MASCP.Service.prototype._retrieveIE = function(dataHash)
323 {
324     // Use XDR
325     var xdr = new XDomainRequest();
326     var loaded = false;
327     var counter = 0;
328     xdr.onerror = dataHash['error'];
329     xdr.open("post",dataHash['url']);
330     xdr.onload = function() {
331         loaded = true;
332         if (dataHash.dataType == 'xml') {
333             var dom = new ActiveXObject("Microsoft.XMLDOM");
334             dom.async = false;
335             dom.loadXML(xhr.responseText,200);
336             dataHash.success(dom, 'success');
337         } else if (dataHash.dataType == 'json' ) {
338             dataHash.success(JSON.parse(xdr.responseText),'success');
339         } else {
340             dataHash.success(xdr.responseText, 'success');
341         }
342     };
343     
344     // We can't set the content-type on the parameters here to url-encoded form data.
345     
346     xdr.send(jQuery.param(dataHash['data']));
347     
348     while (! dataHash.async && ! loaded && counter < 3) {
349         alert("This browser does not support synchronous requests, click OK while we're waiting for data");
350         counter += 1;
351     }
352     if ( ! dataHash.async && ! loaded ) {
353         alert("No data");
354     }
355 };
