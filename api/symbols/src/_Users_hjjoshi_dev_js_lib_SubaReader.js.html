 10 /** Default class constructor
 11  *  @class      Service class that will retrieve data from SUBA for a given AGI.
 12  *              Data is transferred using JSON.
 13  *  @param      {String} agi            Agi to look up
 14  *  @param      {String} endpointURL    Endpoint URL for this service
 15  *  @extends    MASCP.Service
 16  */
 17 MASCP.SubaReader = MASCP.buildService(function(data) {
 18                         jQuery.extend(this,MASCP.SUBA_FIELDS);
 19 
 20                         for (var result_field in MASCP.SUBA_FIELDS) {
 21                             if (data && data.rows && data.rows.length > 0) {
 22                                 this[result_field] = data.rows[0][result_field];
 23                             }
 24                         }
 25                         return this;
 26                     });

 28 MASCP.SubaReader.prototype.requestData = function()
 29 {
 30     var data = [{'type':'condition','sql':{'column':'locus','operator':'in','value':this.agi}}];
 31     return {
 32         type: "POST",
 33         dataType: "json",
 34         data: { 'where' : data.toJSON ? data.toJSON() : JSON.stringify(data),
 35                 'sort'  : 'locus',
 36                 'start' : '0',
 37                 'limit' : '50',
 38                 'dir'   : 'ASC',
 39                 'db'    : 'suba',
 40                 'action': 'db_filter',
 41                 'table' : 'suba2',
 42                 'service' : 'suba' 
 43         }
 44     };
 45 };

 89 MASCP.SubaReader.Result.prototype._parseLocalisation = function(localisation)
 90 {
 91     if (localisation == null || localisation.length == 0 )
 92     {
 93         return null;
 94     }
 95     var experiments = localisation.split(';');
 96     var tissues = {}
 97     for (var i = 0; i < experiments.length; i++) {
 98         var data = experiments[i].split(':');
 99         tissues[data[0]] = tissues[data[0]] || [];
100         tissues[data[0]].push(data[1]);
101     }
102     return tissues;
103 };
104 
105 MASCP.SubaReader.Result.prototype._sortLocalisation = function(loc_data)
106 {
107     var loc_keys = [];
108     for (var i in loc_data) {
109         loc_keys.push(i);
110     }
111     
112     loc_keys = loc_keys.sort(function(a,b) {
113         return loc_data[a].length - loc_data[b].length;
114     });
115     
116     return loc_keys;    
117 };

136 MASCP.SubaReader.Result.prototype.mapController = function(inputElement)
137 {
138     if ( ! this._map ) {
139         return null;
140     }
141     var map = this._map;
142     inputElement = inputElement ? jQuery(inputElement) : jQuery('<div><div class="ms"><input class="ms" type="checkbox"/> MS</div><div class="gfp"><input class="gfp" type="checkbox"/> GFP</div></div>');
143     
144     if ( ! this.getMassSpecLocalisation() )  {
145         jQuery('div.ms', inputElement).css({ 'display': 'none' });
146     } else {
147         var ms_loc = this._sortLocalisation(this.getMassSpecLocalisation());
148         jQuery('input.ms', inputElement).bind('change', function() {
149             if (this.value) {
150                 for (var i in ms_loc) {
151                     map.showKeyword(ms_loc[i], '#ff0000');
152                 }                            
153             } else {
154                 for (var i in ms_loc) {
155                     map.hideKeyword(ms_loc[i], '#ff0000');
156                 }                
157             }
158         }).attr('checked', true);
159     }
160     if ( ! this.getGfpLocalisation() )  {
161         jQuery('div.gfp', inputElement).css({ 'display': 'none' });
162     } else {
163         var gfp_loc = this._sortLocalisation(this.getGfpLocalisation());
164         jQuery('input.gfp', inputElement).bind('change', function() {
165             if (this.value) {
166                 for (var i in gfp_loc) {
167                     map.showKeyword(gfp_loc[i], '#00ff00');
168                 }                            
169             } else {
170                 for (var i in gfp_loc) {
171                     map.hideKeyword(gfp_loc[i], '#00ff00');
172                 }                
173             }
174         }).attr('checked', true);
175     }
176 
177     return inputElement[0];
178 };

180 MASCP.SubaReader.Result.prototype.render = function()
181 {
182     var ms_loc = this._sortLocalisation(this.getMassSpecLocalisation());
183     var gfp_loc = this._sortLocalisation(this.getGfpLocalisation());
184     var container = jQuery('<div><a style="display: block; float: right;" href="http://www.plantenergy.uwa.edu.au/applications/suba/flatfile.php?id='+this.reader.agi+'">SUBA</a></div>')
185     if ( ms_loc.length == 0 && gfp_loc.length == 0 ) {
186         return jQuery('<div>No data</div>');
187     }
188     if (ms_loc.length > 0) {
189         container.append('<div>Proteomics: '+ms_loc.join(', ')+'</div>');
190     }
191     if (gfp_loc.length > 0) {
192         container.append('<div>Flourescent Protein: '+gfp_loc.join(', ')+'</div>');
193     }
194     
195     var do_diagrams = (window.location.search.replace(/^\?/, '').indexOf('drawMap') >= 0);
196     
197     if (typeof GOMap != 'undefined' && do_diagrams) {
198         
199         container.text('');
200         var map = new GOMap.Diagram('cell.svg');
201         var map_container = jQuery('<div style="position: relative; height: 0px; width: 100%; margin-bottom: 2px; overflow: hidden;"></div>');
202         container.append(map_container);
203         map_container.bind('load', function() {
204             for (var i in ms_loc) {
205                 map.showKeyword(ms_loc[i]);
206             }            
207             for (var i in gfp_loc) {
208                 map.showKeyword(gfp_loc[i],'#00ff00');
209             }
210             map_container.css({'height': '100%','overflow':'visible'});
211             map.makeInteractive();
212         });
213         map.appendTo(map_container[0]);
214         container.append('<div style="height: 0px; width: 100%; clear: both; float: none;"></div>');
215         this._map = map;
216     }
217     return container;
218 };
