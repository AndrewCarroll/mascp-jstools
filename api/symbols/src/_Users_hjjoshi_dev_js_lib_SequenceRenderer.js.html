 19 /**
 20  * Register a group with metadata for all sequence renderers.
 21  * @static
 22  * @param {String} groupName    Name to give to this group
 23  * @param {Hash} options        Options to apply to this group - see MASCP.Group for all the fields
 24  * @returns New group object
 25  * @type MASCP.Group
 26  * @see MASCP.event:groupRegistered
 27  * @see MASCP.Group
 28  */
 29 MASCP.registerGroup = function(groupName, options)
 30 {
 31     if ( ! this.groups ) {
 32         this.groups = {};
 33     }
 34     if (this.groups[groupName]) {
 35         return;
 36     }
 37     
 38     var group = new MASCP.Group();
 39     
 40     group['name'] = groupName;
 41     
 42     options = options || {};
 43     
 44     if (options['hide_member_controllers']) {
 45         group['hide_member_controllers'] = true;
 46     }
 47 
 48     if (options['hide_group_controller']) {
 49         group['hide_group_controller'] = true;
 50     }
 51 
 52     if (options['fullname']) {
 53         group['fullname'] = options['fullname'];
 54     }
 55     
 56     if (options['color']) {
 57         group['color'] = options['color'];
 58     }
 59 
 60     group._layers = [];
 61 
 62     group.group_id = new Date().getMilliseconds();
 63     
 64     this.groups[groupName] = group;
 65     
 66     jQuery(MASCP).trigger('groupRegistered',[group]);
 67     
 68     return group;
 69 };

 80 /**
 81  * Register a layer with metadata for all sequence renderers.
 82  * @static
 83  * @param {String} layerName    Name to give to this layer
 84  * @param {Hash} options        Options to set field values for this layer - see the fields for MASCP.Layer.
 85  * @returns New layer object
 86  * @type MASCP.Layer
 87  * @see MASCP.Layer
 88  * @see MASCP.event:layerRegistered
 89  */
 90 MASCP.registerLayer = function(layerName, options)
 91 {
 92     if ( ! this.layers ) {
 93         this.layers = {};
 94     }
 95     if (this.layers[layerName]) {
 96         this.layers[layerName].disabled = false;
 97         return;
 98     }
 99     
100     var layer = new MASCP.Layer();
101     
102     layer['name'] = layerName;
103     
104     options = options || {};
105     
106     if (options['fullname']) {
107         layer['fullname'] = options['fullname'];
108     }
109     
110     if (options['color']) {
111         layer['color'] = options['color'];
112     }
113     
114     if (options['group']) {
115         layer['group'] = this.getGroup(options['group']);
116         if ( ! layer['group'] ) {
117             throw "Cannot register this layer with the given group - the group has not been registered yet";
118         }
119         layer['group']._layers.push(layer);
120     }
121     
122     
123     this.layers[layerName] = layer;
124     
125     if (options['css']) {
126         layerCss = options['css'];
127         layerCss = layerCss.replace(/\.inactive/g, '.'+layerName+'_inactive .'+layerName);
128         layerCss = layerCss.replace(/\.tracks\s+\.active/g, '.'+layerName+'_active .track .'+layerName);
129         layerCss = layerCss.replace(/\.active/g, '.'+layerName+'_active .'+layerName);
130         layerCss = layerCss.replace(/\.overlay/g, '.'+layerName+'_overlay');
131         jQuery('<style type="text/css">'+layerCss+'</style>').appendTo('head');
132     }
133     layer.layer_id = new Date().getMilliseconds();
134     
135     jQuery(MASCP).trigger('layerRegistered',[layer]);
136     
137     return layer;
138 };

241 /**
242  * Set the sequence for this renderer. Fires the sequenceChange event when the sequence is set.
243  * @param {String} sequence Sequence to render
244  * @see MASCP.SequenceRenderer#event:sequenceChange
245  */
246 MASCP.SequenceRenderer.prototype.setSequence = function(sequence)
247 {
248     this.sequence = this._cleanSequence(sequence);
249     var sequence_els = [];
250     var renderer = this;
251     var seq_chars = this.sequence.split('');
252     for (var i =0; i < seq_chars.length; i++) {
253         var aa = seq_chars[i];
254         if (aa.match(/[A-Za-z]/)) {
255             sequence_els.push(jQuery('<span>'+aa+'</span>')[0]);
256         }
257     }
258 
259     jQuery(sequence_els).each( function(i) {
260         if ( (i % 10) == 0 && i > 0 && ((i % 50) != 0)) {
261             this.style.margin = '0px 0px 0px 1em';
262         }
263         if ( (i % 50) == 0 && i > 0 ) {
264             if (MASCP.IE7) {
265                 sequence_els[i-1].style.styleFloat = 'none';
266                 sequence_els[i-1].style.width = '1em';
267             }
268             this.style.clear = 'both';
269         }
270         
271         this._index = i;
272         
273         this.style.display = 'block';
274         this.style.cssFloat = 'left';
275         this.style.styleFloat = 'left';
276         this.style.height = '1.1em';
277         this.style.position = 'relative';
278 
279         this.addToLayer = MASCP.SequenceRenderer.addElementToLayer;
280         this.addBoxOverlay = MASCP.SequenceRenderer.addBoxOverlayToElement;
281         this.addToLayerWithLink = MASCP.SequenceRenderer.addElementToLayerWithLink;
282         this._renderer = renderer;
283     });
284     this._sequence_els = sequence_els;   
285     jQuery(this).trigger('sequenceChange');
286 };

288 /**
289  * Color some residues on this residue
290  * @param {Array} indexes Indexes to apply the given color to
291  * @param {String} color Color to use to highlight the residues
292  * @returns ID for the layer that is created
293  * @type String
294  */
295 MASCP.SequenceRenderer.prototype.colorResidues = function(indexes, color) {
296     var layer_id = Math.floor(Math.random()*1000).toString();
297     MASCP.registerLayer(layer_id, { 'color' : (color || '#ff0000') });
298     var aas = this.getAminoAcidsByPosition(indexes);
299     for (var i = 0; i < aas.length; i++ ) {
300         aas[i].addToLayer(layer_id);
301     }
302     return MASCP.getLayer(layer_id);
303 };

306 MASCP.SequenceRenderer.prototype._cleanSequence = function(sequence) {
307     if ( ! sequence ) {
308         return sequence;
309     }
310     var cleaned_sequence = sequence;
311     cleaned_sequence = cleaned_sequence.replace(new RegExp(String.fromCharCode(160),"g"),'');
312     cleaned_sequence = cleaned_sequence.replace(/[\n\t\s\d]+/mgi,'');
313     cleaned_sequence = cleaned_sequence.replace(/\(.*\)/g,'');
314     return cleaned_sequence;
315 };

317 /**
318  * Retrieve the HTML Elements that contain the amino acids at the given positions. The first amino acid is found at position 1.
319  * @param {Array} indexes Indexes to retrieve elements for
320  * @returns Elements representing each amino acid at the given positions
321  * @type Array
322  */
323 MASCP.SequenceRenderer.prototype.getAminoAcidsByPosition = function(indexes) {
324     var sequence_els = this._sequence_els;
325     return jQuery.map(indexes, function(index) {
326         return sequence_els[index-1];
327     });
328 };

330 /**
331  * Retrieve the HTML Elements that contain the amino acids contained in the given peptide sequence.
332  * @param {String} peptideSequence Peptide sequence used to look up the amino acids
333  * @returns Elements representing each amino acid at the given positions
334  * @type Array
335  */
336 MASCP.SequenceRenderer.prototype.getAminoAcidsByPeptide = function(peptideSequence) {
337     var start = this.sequence.indexOf(peptideSequence);
338     var results = [];
339     if (start < 0) {
340         return results;
341     }
342     for (var i = 0; i < peptideSequence.length; i++) {
343         results.push(this._sequence_els[start+i]);
344     }
345     return results;
346 };

423 /**
424  * Hide or show a group. Fires an event when this method is called.
425  * @param {Object} grp Group to set the visibility for
426  * @param {Boolean} visibility True for visible, false for hidden
427  * @see MASCP.Group#event:visibilityChange
428  */
429 MASCP.SequenceRenderer.prototype.setGroupVisibility = function(grp,visibility) {
430     var groupName = grp;
431     var group;
432     if (typeof grp != 'string') {
433         groupName = group.name;
434         group = grp;
435     } else {
436         group = MASCP.groups[grp];
437         groupName = group.name;
438     }
439     var renderer = this;
440     jQuery(group._layers).each(function(i) {
441         if (this.disabled && visibility) {
442             return;
443         }
444         if (visibility == true) {
445             renderer.showLayer(this.name);
446         } else if (visibility == false) {
447             renderer.hideLayer(this.name);                
448         } else {
449             renderer.toggleLayer(this.name);
450         }
451     });
452     if (visibility != null) {
453         jQuery(group).trigger('visibilityChange',[renderer,visibility]);
454     }
455 };

497 /**
498  * Create a layer controller for this sequence renderer. Attach the controller to the containing box, and shift the box across 20px.
499  */
500 MASCP.SequenceRenderer.prototype.createLayerController = function() {
501     var controller_box = jQuery('<div style="position: absolute; top: 0px; font-family: Helvetica, Arial, Sans-serif; margin-left: 20px; left: 100%; width: 250px;"></div>');
502     var container = jQuery(this._container);
503     container.append(controller_box);
504 
505     if ( ! this._controllers ) {
506         this._controllers = [];
507     }
508     this._controllers.push(controller_box);
509 
510     var renderer = this;
511     
512     jQuery(MASCP).bind('layerRegistered', function(e) {
513 		jQuery(controller_box).accordion('destroy');
514 		jQuery(controller_box).accordion({header : 'h3', collapsible : true, autoHeight: true, active: false });
515 	});
516     
517     
518     controller_box.add_layer = function(layer) {
519         var layer_controller = jQuery('<div><input type="checkbox"/>'+layer.fullname+'</div>');
520         if (layer.group) {
521             jQuery(layer.group._layer_container).append(layer_controller);
522         } else {
523             jQuery(this).append(layer_controller);
524         }
525         
526         renderer.createLayerCheckbox(layer,jQuery('input',layer_controller)[0]);
527     };
528 
529     controller_box.add_group = function(group) {
530         var layer_controller = jQuery('<h3><input style="margin-left: 25px;" type="checkbox"/>'+group.fullname+'</h3>');
531         jQuery(this).append(layer_controller);
532         var children_container = jQuery('<div style="max-height: 200px; overflow: auto;"></div>');
533         jQuery(this).append(children_container);
534         
535         group._layer_container = children_container[0];
536         group._controller = layer_controller;
537         
538         group._check_intermediate = function() {
539             var checked = 0;
540             jQuery(group._layers).each(function(i) {
541                 if (renderer.isLayerActive(this.name)) {
542                     checked++;
543                 }
544             });
545             var input_el = jQuery('input',layer_controller)[0];
546             input_el.indeterminate = (checked != 0 && checked != group._layers.length);
547             if (! input_el.indeterminate ) {
548                 input_el.checked = (checked != 0);
549             }
550         };
551         
552         jQuery(jQuery('input',layer_controller)[0]).bind('mouseup',function(e) {
553             if (this.indeterminate) {
554                 jQuery(this).trigger('change');
555             }
556         });
557 
558         renderer.createGroupCheckbox(group,jQuery('input',layer_controller)[0]);
559         
560     };
561 
562     
563     jQuery(MASCP).bind("layerRegistered",function(e,layer) {
564         if (layer.group && layer.group.hide_member_controllers) {
565             return;
566         }
567         controller_box.add_layer(layer);
568     });
569 
570     jQuery(MASCP).bind("groupRegistered",function(e,group) {
571         if (group.hide_group_controller) {
572             return;
573         }
574         controller_box.add_group(group);
575     });
576 
577     
578     if (MASCP.layers) {
579         for (var layerName in MASCP.layers) {
580             var layer = MASCP.layers[layerName]
581             if (layer.group && layer.group.hide_member_controllers) {
582                 continue;
583             }
584             controller_box.add_layer(layer); 
585         };
586     }
587     return this;
588 };

600 /**
601  * Create a checkbox that is used to control the given layer
602  * @param {String|Object} layer Layer name or layer object that a controller should be generated for
603  * @param {Object} inputElement Optional input element to bind events to. If no element is given, a new one is created.
604  * @returns Checkbox element that when checked will toggle on the layer, and toggle it off when unchecked
605  * @type Object
606  */
607 MASCP.SequenceRenderer.prototype.createLayerCheckbox = function(layer,inputElement) {
608     var renderer = this;
609     if (! MASCP.layers[layer]) {
610         return;
611     }
612 
613     var layerObj = null;
614     
615     if (typeof layer == 'string' && MASCP.layers ) {
616         layerObj = MASCP.layers[layer];
617     } else if (typeof layer == 'object') {
618         layerObj = layer;
619     }
620 
621     if ( ! layerObj ) {
622         return;
623     }
624     
625     
626     var the_input = inputElement || jQuery('<input type="checkbox" value="true"/>')[0];
627     
628     the_input._current_bindings = the_input._current_bindings || [];
629     
630     for (var i = 0; i < the_input._current_bindings.length; i++) {
631         if (    the_input._current_bindings[i].layer == layer && 
632                 the_input._current_bindings[i].renderer == renderer ) {
633             return;
634         }
635     }
636     
637     the_input.removeAttribute('checked');
638     the_input.checked = this.isLayerActive(layer);
639 
640     if (layerObj && the_input._current_bindings.length == 0) {
641         jQuery(layerObj).bind("visibilityChange",function(e,rend,visibility) {
642             if (rend != renderer) {
643                 return;
644             }
645             if (visibility) {
646                 the_input.checked = visibility;
647             } else {
648                 the_input.checked = false;
649                 the_input.removeAttribute('checked');
650             }
651         });
652         if (the_input.parentNode) {
653             the_input.parentNode.insertBefore(jQuery('<div style="position: relative; left: 0px; top: 0px; float: left; background-color: '+layerObj.color+'; width: 1em; height: 1em;"></div>')[0],the_input);
654         }
655     }
656 
657     
658     the_input._current_bindings.push({ 'layer' : layer , 'renderer' : renderer });
659 
660     jQuery(the_input).bind('change',function(e) {
661         if (this.checked) {
662             renderer.showLayer(layer,false);
663         } else {
664             renderer.hideLayer(layer,false);
665         }
666         if (MASCP.getLayer(layer).group && MASCP.getLayer(layer).group._check_intermediate) {
667             MASCP.getLayer(layer).group._check_intermediate();
668         }
669     });
670     
671     return the_input;    
672 };

698     }
699     return (typeof group == 'string') ? MASCP.groups[group] : group;
700 };

702 /**
703  * Create a checkbox that is used to control the given group
704  * @param {String|Object} group Group name or group object that a controller should be generated for
705  * @param {Object} inputElement Optional input element to bind events to. If no element is given, a new one is created.
706  * @returns Checkbox element that when checked will toggle on the group, and toggle it off when unchecked
707  * @type Object
708  */
709 MASCP.SequenceRenderer.prototype.createGroupCheckbox = function(group,inputElement) {
710     var renderer = this;
711     var the_input = inputElement ? jQuery(inputElement) : jQuery('<input type="checkbox" value="true"/>');
712     var groupObject = MASCP.getGroup(group);
713     
714     if (! groupObject ) {
715         return;
716     }
717 
718     the_input[0]._current_bindings = the_input[0]._current_bindings || [];
719     
720     for (var i = 0; i < the_input[0]._current_bindings.length; i++) {
721         if (    the_input[0]._current_bindings[i].group == group && 
722                 the_input[0]._current_bindings[i].renderer == renderer ) {
723             return;
724         }
725     }
726     
727     the_input[0].removeAttribute('checked');
728     the_input.bind('change',function(e) {        
729         group_obj = MASCP.getGroup(group);
730         if (! group_obj ) {
731             return;
732         }
733         if (this.checked) {
734             jQuery(group_obj._layers).each(function(i) {
735                 renderer.showLayer(this.name,false);
736             });
737         } else {
738             jQuery(group_obj._layers).each(function(i) {
739                 renderer.hideLayer(this.name,false);
740             });                
741         }
742     });
743 
744 
745     if (groupObject && the_input[0]._current_bindings.length == 0) {
746         jQuery(MASCP.getGroup(group)).bind('visibilityChange', function(e,rend,visibility) {
747             if (rend != renderer) {
748                 return;
749             }
750             the_input[0].checked = visibility;
751             if ( ! visibility ) {
752                 the_input[0].removeAttribute('checked');
753             }
754         });
755         if (the_input[0].parentNode) {
756             the_input[0].parentNode.insertBefore(jQuery('<div style="position: relative; left: 0px; top: 0px; float: left; background-color: '+groupObject.color+'; width: 1em; height: 1em;"></div>')[0],the_input[0]);
757         }
758     }
759 
760     the_input[0]._current_bindings.push({ 'group' : group , 'renderer' : renderer });
761 
762     the_input.bind('click',function(e) {
763         e.stopPropagation();
764     });
765     
766     return the_input;
767 };

823 MASCP.SequenceRenderer.prototype.reset = function()
824 {
825     jQuery(this._container).attr('class',null);

827         this.hideGroup(group);

829     for ( var layer in MASCP.layers) {

832     }
