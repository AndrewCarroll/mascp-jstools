 15                         this._raw_data = data;                        
 16                         return this;
 17                     });

 19 MASCP.PromexReader.prototype.requestData = function()
 20 {
 21     var agi = (this.agi+"").replace(/\..*$/,'');
 22     
 23     return {
 24         type: "POST",
 25         dataType: "xml",
 26         data: { 'action'    : 11,
 27                 'txt_query' : agi,
 28                 'service'   : 'promex' 
 29         }
 30     };
 31 };

 40 /** Retrieve the URLS for the spectra from the Promex service
 41  *  @returns    Array of urls for the spectra for this entry
 42  *  @type [String]
 43  */
 44 MASCP.PromexReader.Result.prototype.getSpectraURLs = function()
 45 {
 46     var bleh = this._raw_data;
 47     var content = null;
 48     if (this._raw_data.getElementById) {
 49         content = this._raw_data.getElementById('contentArea');
 50     } else {
 51         content = this._raw_data.selectSingleNode("//*[@id = 'contentArea']");
 52     }
 53     var imgs = content.getElementsByTagName('img');
 54     var urls  = [];
 55     for (var i = 0; i < imgs.length; i++) {
 56         urls.push(imgs[i].getAttribute('src'));
 57     }
 58     return urls;
 59 };

 62 /** Retrieve the peptides for this particular entry from the Promex service
 63  *  @returns Array of peptide strings
 64  *  @type [String]
 65  */
 66 MASCP.PromexReader.Result.prototype.getPeptides = function()
 67 {
 68     var content = null;
 69     
 70     if (this._raw_data.getElementById) {
 71         content = this._raw_data.getElementById('contentArea');
 72     } else {
 73         content = this._raw_data.selectSingleNode("//*[@id = 'contentArea']");
 74     }
 75     var tds = jQuery(content.getElementsByTagName('td'));
 76     var peptides = [];
 77     var result = this;
 78     tds.each(function(i) {
 79         if (jQuery(this).text() == 'peptide sequence') {
 80             peptides.push(result._cleanSequence(jQuery(tds[i+1]).text()));            
 81         }
 82     });
 83     return peptides;
 84 };
 85 
 86 MASCP.PromexReader.Result.prototype._cleanSequence = function(sequence)
 87 {
 88     return sequence.replace(/[^A-Z]/g,'');
 89 };

 91 MASCP.PromexReader.prototype.setupSequenceRenderer = function(sequenceRenderer)
 92 {
 93 
 94     MASCP.registerGroup('promex_experimental', {'fullname' : 'ProMex spectra data', 'hide_member_controllers' : true, 'hide_group_controller' : true, 'color' : '#ff9900' });
 95 
 96     this.bind('resultReceived', function() {
 97         var spectra = this.result.getSpectraURLs();
 98         // var css_block = '.active { background: #ff9900; color: #ffffff;} :indeterminate { background: #ff0000; } .active a:hover { background: transparent !important; } .inactive { }';
 99         var css_block = '.active { text-decoration: underline; color: #000000; } .active a { color: #000000; }  :indeterminate { background: #ff0000; } .tracks .active { background: #0000ff; } .inactive a { text-decoration: none; } .inactive { display: none; }';
100     	jQuery(this.result.getPeptides()).each(function(i) {
101             
102         	MASCP.registerLayer('promex_experimental_spectrum_'+i, { 'fullname': 'ProMex spectrum', 'group' : 'promex_experimental', 'color' : '#ff9900', 'css' : css_block });
103     	    var a_spectra = spectra[i];
104     	    var peptide = this;
105     	    var peptide_bits = sequenceRenderer.getAminoAcidsByPeptide(peptide);
106     	    var layer_name = 'promex_experimental_spectrum_'+i;
107     	    peptide_bits[0].addToLayerWithLink(layer_name,a_spectra,peptide_bits.length);
108     	    jQuery(MASCP.getLayer('promex_experimental_spectrum_'+i)).bind('click',function() {
109     	        window.open(a_spectra);
110     	    });
111     	});
112     })
113     return this;
114 };

116 MASCP.PromexReader.Result.prototype.render = function()
117 {
118     if (this.getSpectraURLs().length > 0) {
119         var params = jQuery.param(this.reader.requestData()['data']);
120         var a_container = jQuery('<div>MS/MS spectra <input class="group_toggle" type="checkbox"/><a style="display: block; float: right;" href="http://www.promexdb.org/cgi-bin/peplib.pl?'+params+'">ProMEX</a></div>');
121         jQuery(this.reader.renderers).each(function(i){
122             this.createGroupCheckbox('promex_experimental',jQuery('input.group_toggle',a_container));
123         });
124         return a_container;
125     } else {
126         return null;
127     }
