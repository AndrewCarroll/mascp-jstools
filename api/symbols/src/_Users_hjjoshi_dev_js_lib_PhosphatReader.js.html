  7 /** Default class constructor
  8  *  @class      Service class that will retrieve data from Phosphat for a given AGI.
  9  *              Data is transferred using the JSON format.
 10  *  @param      {String} agi            Agi to look up
 11  *  @param      {String} endpointURL    Endpoint URL for this service
 12  *  @extends    MASCP.Service
 13  */
 14 MASCP.PhosphatReader =  MASCP.buildService(function(data) {
 15                             if (data && data.request_method == 'getPredictedAa') {
 16                                 this._raw_data = data;
 17                             }
 18                             if (data && data.request_method == 'getExperimentsModAa') {
 19                                 this._raw_experimental_data = data;
 20                             }
 21                             return this;
 22                         });

 24 MASCP.PhosphatReader.prototype.requestData = function()
 25 {
 26     var data = [null,this.agi];
 27     
 28     var method = this._method ? this._method : 'getPredictedAa';
 29 
 30     var self = this;
 31     
 32     return {
 33         type: "POST",
 34         dataType: "json",
 35         data: { 'id'        : 1,
 36                 'method'    : method,
 37                 'params'    : data.toJSON ? data.toJSON() : JSON.stringify(data),
 38                 'service'   : 'phosphat' 
 39         },
 40         success: function(data,status) {
 41             data.request_method = method;
 42             self._dataReceived(data,status);
 43             if (self.result && self.result._raw_data && self.result._raw_experimental_data) {
 44                jQuery(self).trigger('resultReceived'); 
 45             }
 46         }
 47     };
 48 };

 52 MASCP.PhosphatReader.prototype.retrieve = function()
 53 {
 54     this._single_retrieve();
 55     this._method = 'getExperimentsModAa';
 56     this._single_retrieve();

 67 /** Retrieve an array of positions that phosphorylation has been predicted to occur upon
 68  *  @returns {Array}    Phosphorylation positions upon the full protein
 69  */
 70 MASCP.PhosphatReader.Result.prototype.getAllPredictedPositions = function()
 71 {
 72     var positions = []; 
 73     for ( var prediction_idx in this._raw_data.result ) {
 74         var prediction = this._raw_data.result[prediction_idx];
 75         if (prediction.prd_score > 0) {
 76             positions.push(prediction.prd_position);
 77         }
 78     }
 79     return positions;
 80 };
 81 
 82 /** Retrieve an array of positions that phosphorylation has been experimentally verified to occur upon
 83  *  @returns {Array}    Phosphorylation positions upon the full protein
 84  */
 85 MASCP.PhosphatReader.Result.prototype.getAllExperimentalPositions = function()
 86 {
 87     var exp_sites = {};
 88     for ( var site_idx in this._raw_experimental_data.result ) {
 89         var site = this._raw_experimental_data.result[site_idx];
 90         if (site['modificationType'] != 'phos' && site['modificationType'] != 'phos_amb') {
 91             continue;
 92         }
 93         var site_id = site['prot_sequence'].indexOf(site['pep_sequence']);
 94         if (site_id < 0) {
 95             continue;
 96         }
 97         site_id += site['position'];
 98         exp_sites[site_id] = 1;
 99     }
100     var positions = [];
101     for ( var i in exp_sites ) {
102         positions.push(parseInt(i));
103     }
104     return positions;
105 };

108 MASCP.PhosphatReader.Result.prototype.render = function()
109 {
110     var result_block;
111     
112     if (this.getAllExperimentalPositions().length > 0) {
113         result_block = jQuery(' <div>Phosphorylation (STY) \
114                                     <input type="checkbox" class="exp_toggle"/> <a style="display: block; float: right;" href="http://phosphat.mpimp-golm.mpg.de/app.html?agi='+this.reader.agi+'">PhosPhAt</a> \
115                                 </div>');
116     } else if (this.getAllPredictedPositions().length > 0) {
117         result_block = jQuery(' <div>Theoretical phosphorylaytion <input type="checkbox" class="theoretical_toggle"/> \
118                                     <a style="display: block; float: right;" href="http://phosphat.mpimp-golm.mpg.de/app.html?agi='+this.reader.agi+'">PhosPhAt</a> \
119                                 </div>');
120     } else {
121         result_block = jQuery(' <div>No phosphorylation \
122                                     <a style="display: block; float: right;" href="http://phosphat.mpimp-golm.mpg.de/app.html?agi='+this.reader.agi+'">PhosPhAt</a> \
123                                 </div>');
124     }
125     
126     jQuery( this.reader.renderers ).each( function(i){
127         this.createLayerCheckbox('phosphat_experimental',jQuery('input.exp_toggle',result_block)[0]);
128         this.createLayerCheckbox('phosphat_theoretical',jQuery('input.theoretical_toggle',result_block)[0]);
129     });
130     return result_block;
131 };

133 MASCP.PhosphatReader.prototype.setupSequenceRenderer = function(sequenceRenderer)
134 {
135     var reader = this;
136     
137     MASCP.registerGroup('phosphat', {'fullname' : 'PhosPhAt data', 'hide_member_controllers' : true, 'hide_group_controller': true });
138 	MASCP.registerLayer('phosphat_theoretical', { 'fullname': 'PhosPhAt theoretical data', 'group': 'phosphat', 'color' : '#3D907B', 'css' : '.active { background: #3D907B; color: #ffffff; } .tracks .active { background: #3D907B; fill: #3D907B; } .inactive { display: none; }' });
139 	MASCP.registerLayer('phosphat_experimental', { 'fullname': 'PhosPhAt experimental data', 'group':'phosphat', 'color' : '#000000', 'css' : '.active { color: #000000; font-weight: bolder; } .tracks .active { background: #000000; fill: #000000; } .inactive { display: none; }' });
140 
141     this.bind('resultReceived', function() {
142         jQuery(sequenceRenderer.getAminoAcidsByPosition(this.result.getAllPredictedPositions())).each(function(i) {
143             MASCP.registerGroup('phosphat', {'fullname' : 'PhosPhAt data', 'hide_member_controllers' : false });
144         	MASCP.registerLayer('phosphat_theoretical', { 'fullname': 'PhosPhAt theoretical data', 'group': 'phosphat', 'color' : '#3D907B', 'css' : '.active { background: #3D907B; color: #ffffff; } .tracks .active { background: #3D907B; fill: #3D907B; } .inactive { display: none; }' });
145             this.addToLayer('phosphat_theoretical');
146         });
147         jQuery(sequenceRenderer.getAminoAcidsByPosition(this.result.getAllExperimentalPositions())).each(function(i) {
148             MASCP.registerGroup('phosphat', {'fullname' : 'PhosPhAt data', 'hide_member_controllers' : false });
149         	MASCP.registerLayer('phosphat_experimental', { 'fullname': 'PhosPhAt experimental data', 'group':'phosphat', 'color' : '#000000', 'css' : '.active { color: #000000; font-weight: bolder; } .tracks .active { background: #000000; fill: #000000; } .inactive { display: none; }' });
150             this.addToLayer('phosphat_experimental');
151         });
152         jQuery(sequenceRenderer).trigger('resultsRendered',[reader]);
153     })
154     return this;
155 };
