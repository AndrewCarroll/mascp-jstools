 50 /**
 51  * @class       A diagram that can be marked up with keywords.
 52  * @param       image   Image to be used for the diagram. Either an url to an svg file, an existing object element with a src attribute, or a reference to an SVG element if the SVG has been inlined.
 53  * @author      hjjoshi
 54  * @requires    svgweb
 55  */
 56 GOMap.Diagram = function(image) {
 57 
 58     this._highlighted = {};
 59     this._styles_cache = {};
 60     var url = null;
 61     if (typeof image == 'string') {
 62         url = image;
 63         image = null;
 64     } else if (image.nodeName && image.nodeName.toLowerCase() == 'object') {
 65         url = image.getAttribute('src') || image.getAttribute('data');
 66     } else if (image.nodeName && image.nodeName.toLowerCase() == 'svg') {
 67         this.element = image;
 68         this._svgLoaded();
 69         if (image.fireEvent) {
 70             var ev = document.createEventObject();
 71             ev.type = 'load';
 72             document.body.fireEvent("onreadystatechange",ev);
 73         } else {
 74             var evt = document.createEvent('Events');
 75             evt.initEvent('load',false,true);
 76             image.dispatchEvent(evt);
 77         }
 78         return;
 79     }
 80 
 81     
 82     this.element = document.createElement('object',true);
 83     
 84     this.element.setAttribute('data',url);
 85     this.element.setAttribute('type','image/svg+xml');
 86     this.element.setAttribute('width','100%');
 87     this.element.setAttribute('height','100%');
 88 
 89     var self = this;
 90     this.element.addEventListener('load',function() {
 91         self.element = (this.contentDocument || this.getAttribute('contentDocument')).rootElement;
 92         
 93         // Make the destroy function an anonymous function, so it can access this new
 94         // element without having to store it in a field
 95         
 96         var svg_object = this;
 97         self.destroy = function() {
 98             svgweb.removeChild(svg_object, svg_object.parentNode);
 99         };
100 
101         self._svgLoaded();
102         if (self._container) {
103             var evt = document.createEvent('Events');
104             evt.initEvent('load',false,true);
105             self._container.dispatchEvent(evt);
106         }
107     },false);
108     
109     if (image) {
110         this.appendTo(image.parentNode);
111         image.parentNode.removeChild(image);
112     }
113 };

132 /**
133  * Highlight a given keyword on the diagram
134  * @param {String} keyword  GO keyword to highlight
135  * @param {String} color    CSS color string to use as the highlighting colour. Defaults to #ff0000.
136  */
137 GOMap.Diagram.prototype.showKeyword = function(keyword,color) {
138     var els = this._elementsForKeyword(keyword);
139     
140     if (this._highlighted[keyword] && ! color) {
141         return;
142     }
143     
144     color = color || '#ff0000';
145     
146     this._highlighted[keyword] = true;
147 
148     for (var i = 0; i < els.length; i++ ) {
149         els[i]._highlighted = els[i]._highlighted || {};
150         els[i]._highlighted[color] = true;
151         if (els[i].nodeName == 'path' || els[i].nodeName == 'circle') {
152             this._outlineElement(els[i]);
153         }
154     }
155     var self = this;
156     this._recurse(els, function(el) {
157         self._highlightElement(el);
158     });
159         
160 };

171 /**
172  * Hide a given keyword on the diagram
173  * @param {String} keyword  GO keyword to turn highlighting off for
174  */
175 GOMap.Diagram.prototype.hideKeyword = function(keyword,color) {
176     var els = this._elementsForKeyword(keyword);
177     var self = this;
178 
179     this._highlighted[keyword] = false;
180     
181     this._recurse(els, function(el) {
182         if (color != null && el._highlighted) {
183             el._highlighted[color] = false;
184         } else {
185             el._highlighted = {};
186         }
187         for (var col in el._highlighted) {
188             if (el._highlighted[col] == true) {
189                 self._outlineElement(el);
190                 return;
191             }
192         }
193         self._restoreStyle(el);
194     });
195 };

302 /*
303  * Retrieve all the SVG elements that match the given keyword. The SVG document
304  * should have elements marked up with a keyword attribute.
305  * @param {String}  keyword     Keyword to use to search for elements
306  */
307 GOMap.Diagram.prototype._elementsForKeyword = function(keyword) {
308     var root_svg = this.element;
309     var els = [];
310 
311     if (! GOMap.IE ) {
312         els = this._execute_xpath(root_svg,"//*[@keyword = '"+keyword+"']",root_svg.ownerDocument);
313     } else {
314         var some_els = root_svg.ownerDocument.getElementsByTagNameNS(svgns,'*');
315         for (var i = 0; i < some_els.length; i++) {
316             var el_key = some_els[i].getAttribute('keyword');
317             if (el_key == keyword) {                
318                 els.push(some_els[i]);
319             }
320         }
321     }
322     return els;
323 };

325 /* 
326  * Execute an xpath query upon a document, pulling the results into an array of elements
327  * @param {Element} element Start element
328  * @param {String} xpath Xpath query to execute
329  * @param {Document} document Parent document
330  */
331 GOMap.Diagram.prototype._execute_xpath = function(element, xpath, doc) {
332     var results = [];
333     if (doc.evaluate) {
334         xpath_result = doc.evaluate(xpath,element,null,XPathResult.UNORDERED_NODE_SNAPSHOT_TYPE,null);
335         var i = 0;
336         while ( (a_result = xpath_result.snapshotItem(i)) != null ) {
337             results.push(a_result);
338             i++;
339         }
340     } else {
341         xpath_result = element.selectNodes(xpath);
342         for (var i = 0; i < xpath_result.length; i++ ){
343             results[i] = xpath_result.item(i);
344         }
345     }
346     return results;
347 };
348 /*
349  * Perform a breadth-first traversal of the nodelist.
350  * @param {Array} nodelist Starting list of nodes to perform traversal over
351  * @param {Function} function Callback for this traversal. Callback function takes a single argument, which is the currently inspected node.
352  */
353 GOMap.Diagram.prototype._recurse = function(nodelist,callback) {
354     for (var i = 0; i < nodelist.length; i++) {
355         callback.call(this,nodelist[i]);        
356         if (nodelist[i].childNodes.length > 0) {
357             this._recurse(nodelist[i].childNodes,callback);
358         }
359     }
360 };
361 /*
362  * Cache the old style for this element. We need to cache the style for the element so that we can restore the
363  * element style when it is active.
364  * @param {Element} el Element to store the style for
365  */
366 GOMap.Diagram.prototype._cacheStyle = function(el) {    
367     if ( ! el.id ) {
368         var an_id = 'svg'+(new Date).getTime().toString();
369         el.setAttribute('id',an_id);
370     }
371 
372     if (this._styles_cache[el.id] != null || ! el.style || ! el.id ) {
373         return;
374     }
375     this._styles_cache[el.id] = {
376         'stroke'         : el.style.getPropertyValue('stroke'),
377         'stroke-width'   : el.style.getPropertyValue('stroke-width') || '0px',
378         'opacity'        : el.style.getPropertyValue('opacity'),
379         'fill-opacity'   : el.style.fillOpacity || el.style.getPropertyValue('fill-opacity'),
380         'stroke-opacity' : el.style.strokeOpacity || el.style.getPropertyValue('stroke-opacity')
381     };
382 };
383 
384 /*
385  * Restore the style for an element from the cache
386  * @param {Element} element Element to restore the style for. This element must have cache data.
387  */
388 
389 GOMap.Diagram.prototype._restoreStyle = function(element) {
390     if ( ! element.style ) {
391         return;
392     }
393     if (this._styles_cache[element.id]) {
394         var cacher = this._styles_cache[element.id];
395         var properties = {'stroke-width':null,'opacity':null,'stroke':null,'fill-opacity':null,'stroke-opacity':null};
396         for (var prop in properties) {
397             // We don't set null properties in IE because they cause the wrong styles to be displayed
398             if ( GOMap.IE && ! cacher[prop] ) {
399                 continue;
400             }
401             element.style.setProperty(prop,cacher[prop],null);
402         }
403     }
404 };

406 /*
407  * Draw an outline around the given element
408  * @param {Element} element Element to outline
409  */
410 GOMap.Diagram.prototype._outlineElement = function(element) {
411     if ( ! element.style ) {
412         return;
413     }
414     this._cacheStyle(element);
415     
416     var target_color = this._calculateColorForElement(element);
417     
418     element.style.setProperty('stroke',target_color,null);
419     element.style.setProperty('stroke-width',4,null);
420     element.style.setProperty('stroke-opacity',1,null);
421 };

423 /*
424  * Calculate the color fill. Since there may be more than one color highlighted
425  * on this element, we build the pattern if needed, and return a reference to 
426  * that pattern if a pattern is to be used.
427  */
428 
429 GOMap.Diagram.prototype._calculateColorForElement = function(element) {
430     var pattern = "pat";
431     var total_keywords = 0;
432     for (var col in element._highlighted) {
433         if (element._highlighted && element._highlighted[col] == true) {
434             pattern += "_"+col;
435             total_keywords++;
436         }
437     }
438     if (total_keywords == 1) {
439         return pattern.split(/_/)[1];
440     } else {
441         return this._buildPattern(pattern);
442     }
443     
444 };
445 /*
446  * Create a pattern element under the defs element for the svg. If there isn't a defs element
447  * there already, create one and append it to the document.
448  * @param {String} pattern_name Underscore separated pattern name - each component of the pattern should be represented. e.g. ff0000_00ff00_0000ff
449  * @returns The color name that can be used to reference this pattern
450  * @type String
451  */
452 GOMap.Diagram.prototype._buildPattern = function(pattern_name) {
453     var pattern_els = pattern_name.split('_');
454     this._cached_patterns = this._cached_patterns || {};
455     var cleaned_name = pattern_name.replace(/#/g,'');
456     if (this._cached_patterns[cleaned_name]) {
457         return 'url(#'+cleaned_name+')';
458     }
459     
460     var root_svg = this.element;
461     var defs_el = root_svg.ownerDocument.getElementsByTagNameNS(svgns,'defs')[0];
462 
463     if ( ! defs_el ) {
464         defs_el = document.createElementNS(svgns,'defs');
465         root_svg.appendChild(defs_el);
466     }
467 
468     var new_pattern = document.createElementNS(svgns,'pattern');
469     new_pattern.setAttribute('x','0');
470     new_pattern.setAttribute('y','0');
471     new_pattern.setAttribute('width','30');
472     new_pattern.setAttribute('height','30');
473     new_pattern.setAttribute('viewBox', '0 0 100 100');
474     new_pattern.setAttribute('patternUnits','userSpaceOnUse');
475     new_pattern.setAttribute('patternTransform','rotate(45)');
476     new_pattern.setAttribute('id',cleaned_name);
477     var pattern_width = 100.0 / (pattern_els.length - 1);
478     var start_pos = 0;
479     
480     for (var i = 1; i < pattern_els.length; i++ ) {
481         var a_box = document.createElementNS(svgns, 'rect');
482         a_box.setAttribute('x', start_pos);
483         start_pos += pattern_width;
484         a_box.setAttribute('y', 0);
485         a_box.setAttribute('width', pattern_width);
486         a_box.setAttribute('height', 100);
487         a_box.setAttribute('fill', pattern_els[i]);
488         new_pattern.appendChild(a_box);
489     }
490 
491 
492     defs_el.appendChild(new_pattern);
493     this._cached_patterns[cleaned_name] = true;
494     return 'url(#'+cleaned_name+')';
495 };

497 /* Highlight an element by making it opaque
498  * @param {Element} element Element to make opaque
499  */
500 GOMap.Diagram.prototype._highlightElement = function(element) {
501     // Skip this if we don't have a style or has no id and isn't a group
502     if (! element.style || (! element.id && ! element.nodeName == 'g') ) {
503         return;
504     }
505     
506     this._cacheStyle(element);
507 
508     if (element.nodeName == 'path' || element.nodeName == 'circle') {
509         element.setAttribute('opacity','1');
510         element.style.setProperty('opacity',1,null);
511     }
512     element.setAttribute('fill-opacity','1');
513     element.setAttribute('stroke-opacity','1');
514     if (element.style.setProperty) {
515         element.style.setProperty('fill-opacity',1,null);
516         element.style.setProperty('stroke-opacity',1,null);
517     }
518 };
519 
520 /* Go through all the elements in the svg document and force the opacity to be translucent. Since
521  * svgweb doesn't support the referencing of extrinsic stylesheets, we need to go through and 
522  * explicitly set the opacity for all the elements. This is really slow on Internet Explorer.
523  * We've got different behaviour for the different svg element types as they all react differently
524  * to having their opacity set.
525  */
526 GOMap.Diagram.prototype._forceOpacity = function() {
527     var root_svg = this.element;
528     var suspend_id = root_svg.suspendRedraw(5000);
529     var els = root_svg.ownerDocument.getElementsByTagNameNS(svgns,'*');
530     for (var i = 0; i < els.length; i++ ) {
531         if (els[i].nodeName == 'svg') {
532             continue;
533         }
534         if (els[i].parentNode && els[i].parentNode.parentNode && els[i].parentNode.parentNode.nodeName == 'defs') {
535             continue;
536         }
537         if (els[i].nodeName == 'defs' || (els[i].parentNode && els[i].parentNode.nodeName == 'defs')) {
538             continue;
539         }
540         if (els[i].nodeName == 'path') {
541             els[i].setAttribute('opacity','0.4');
542             els[i].style.opacity = 0.4;
543         } else {
544             els[i].setAttribute('fill-opacity','0.3');
545             els[i].setAttribute('stroke-opacity','0.2');
546             if (els[i].style) {
547                 els[i].style.fillOpacity = 0.3;
548                 els[i].style.strokeOpacity = 0.2;
549             }
550         }
551     }
552     root_svg.unsuspendRedraw(suspend_id);
553 };

556 /**
557  * @class       State class for adding panning functionality to an element. Each element that is to be panned needs a new instance
558  *              of the Dragger to store state.
559  * @author      hjjoshi
560  * @requires    svgweb
561  */
562 GOMap.Diagram.Dragger = function() {
563   this.oX = 0;
564   this.oY = 0;
565   this.dX = 0;
566   this.dY = 0;
567   this.dragging = false;
568   this.targetElement = null;
569 };
570 
571 /**
572  * Connect this dragger to a particular element. If an SVG element is given, panning occurs within the bounding box of the SVG, and
573  * the image is shifted by using the currentTranslate property. If a regular HTML element is given, the scrollLeft and scrollTop attributes
574  * are used to move the viewport around. 
575  * @param {Element} targetElement Element to enable panning upon.
576  */
577 GOMap.Diagram.Dragger.prototype.applyToElement = function(targetElement) {
578     var self = this;
579     var svgMouseDown = function(evt) {
580       var positions = mousePosition(evt);
581       self.dragging = true;
582       if (self.targetElement) {
583           self.oX = positions[0];
584           self.oY = positions[1];
585           self.dX = self.targetElement.scrollLeft;
586           self.dY = self.targetElement.scrollTop;
587           evt.preventDefault(true);
588           return;
589       }
590 
591       var p = targetElement.createSVGPoint();
592       var positions = mousePosition(evt);
593       p.x = positions[0];
594       p.y = positions[1];
595 
596       var rootCTM = targetElement.getScreenCTM();
597       self.inverseRootCTM = rootCTM.inverse();
598 
599       p = p.matrixTransform(self.inverseRootCTM);
600 
601       var cur_x = targetElement.currentTranslate.x || targetElement.currentTranslate.getX();
602       var cur_y = targetElement.currentTranslate.y || targetElement.currentTranslate.getY();
603 
604       self.oX = p.x - self.dX - cur_x;
605       self.oY = p.y - self.dY - cur_y;
606 
607       evt.preventDefault(true);
608     };
609     
610     var mousePosition = function(evt) {
611         var posx = 0;
612         var posy = 0;
613         if (!evt) var evt = window.event;
614         if (evt.pageX || evt.pageY) 	{
615             posx = evt.pageX;
616             posy = evt.pageY;
617         } else if (evt.clientX || evt.clientY) 	{
618             posx = evt.clientX + document.body.scrollLeft + document.documentElement.scrollLeft;
619             posy = evt.clientY + document.body.scrollTop + document.documentElement.scrollTop;
620         }
621         if (self.targetElement) {
622             posx = evt.screenX;
623             posy = evt.screenY;
624         }
625         return [ posx, posy ];
626     }
627     
628     var mouseMove = function(evt) {
629         this.style.cursor = 'all-scroll';
630         var positions = mousePosition(evt);
631         if (!self.dragging) {
632            return;
633         }
634         this.style.cursor = '-moz-grabbing';
635         targetElement.scrollLeft = self.dX + (self.oX - positions[0]);
636         targetElement.scrollTop = self.dY + (self.oY - positions[1]);
637         evt.preventDefault(true);
638     };
639     
640     var mouseDown = function(evt) {
641         self.dragging = true;
642         var positions = mousePosition(evt);
643         self.oX = positions[0];
644         self.oY = positions[1];
645         self.dX = targetElement.scrollLeft;
646         self.dY = targetElement.scrollTop;
647         evt.preventDefault(true);
648     };
649     
650     var svgMouseMove = function(evt) {
651         this.style.cursor = 'all-scroll';
652         var positions = mousePosition(evt);
653         if (!self.dragging) {
654             return;
655         }
656         this.style.cursor = '-moz-grabbing';        
657         if (self.targetElement) {
658             self.targetElement.scrollLeft = self.dX + (self.oX - positions[0]);
659             self.targetElement.scrollTop = self.dY + (self.oY - positions[1]);
660             evt.preventDefault(true);
661           return;
662         }
663         evt.preventDefault(true);
664 
665 
666         var p = targetElement.createSVGPoint();
667         var positions = mousePosition(evt);
668 
669 
670 
671         p.x = positions[0];
672         p.y = positions[1];
673         
674         p = p.matrixTransform(self.inverseRootCTM);
675         p.x -= self.oX;
676         p.y -= self.oY;
677 
678         self.dX = p.x;
679         self.dY = p.y;
680 
681         if (p.x > 0 || p.y > 0) {
682             return;
683         }
684 
685         if (targetElement.currentTranslate.setXY) {
686             targetElement.currentTranslate.setXY(p.x,p.y);
687         } else {
688             targetElement.currentTranslate.x = p.x;
689             targetElement.currentTranslate.y = p.y;          
690         }
691     };
692 
693     var mouseOut = function(e) {
694         var toTarget = e.relatedTarget || e.toElement;
695         e.preventDefault(true);
696 
697         while (toTarget != null) {
698             if (toTarget == this) {
699                 return;
700             }
701             toTarget = toTarget.parentNode;
702         }
703         self.dragging = false;
704     };
705     
706     var mouseUp = function(evt) { 
707       self.oX = 0;
708       self.oY = 0;
709       self.dX = null;
710       self.dY = null;
711       self.dragging = false;
712       evt.preventDefault(true);
713     };
714     
715     targetElement.setAttribute('cursor','pointer');    
716     
717     if ( ! targetElement.addEventListener) {
718         targetElement.addEventListener = function(name,func,bool) {
719             this.attachEvent(name,func);
720         }
721     }
722 
723     if (targetElement.nodeName == 'svg') {
724         targetElement.addEventListener('mousedown', svgMouseDown, false);
725         targetElement.addEventListener('mousemove', svgMouseMove, false);        
726         targetElement.addEventListener('mouseup', mouseUp, false);
727     } else {
728         targetElement.addEventListener('mousedown', mouseDown, false);
729         targetElement.addEventListener('mousemove', mouseMove, false);        
730         targetElement.addEventListener('mouseup', mouseUp, false);        
731         targetElement.addEventListener('mouseout',mouseOut, false);
732     }
733 
734 };
735 /**
736  * Given an element that implements a zoom attribute, creates a div that contains controls for controlling the zoom attribute. The
737  * zoomElement must have a zoom attribute, and can fire the zoomChange event whenever the zoom value is changed on the object. The
738  * scrollwheel is connected to this element so that when the mouse hovers over the controls, it can control the zoom using only
739  * the scroll wheel.
740  * @param {Object} zoomElement Element to control the zooming for.
741  * @param {Number} min Minimum value for the zoom attribute (default 0)
742  * @param {Number} max Maximum value for the zoom attribute (default 10)
743  * @param {Number} precision Step precision for the zoom control (default 0.5)
744  * @param {Number} value Default value for this control
745  * @returns DIV element containing the controls
746  * @type Element
747  * @see GOMap.Diagram#event:zoomChange
748  */
749 GOMap.Diagram.addZoomControls = function(zoomElement,min,max,precision,value) {
750     min = min || 0;
751     max = max || 10;
752     precision = precision || 0.5;
753     value = value || min; 
754     
755     var controls_container = document.createElement('div');
756     controls_container.style.height = '30%';   
757     var zoomIn = document.createElement('input');
758     zoomIn.setAttribute('type','button');
759     zoomIn.setAttribute('value','+');
760     var zoomOut = document.createElement('input');
761     zoomOut.setAttribute('type','button');
762     zoomOut.setAttribute('value','-');
763     var reset = document.createElement('input');
764     reset.setAttribute('type','button');
765     reset.setAttribute('value','Reset');
766 
767     controls_container.appendChild(reset);    
768 
769     reset.addEventListener('click',function() {
770         zoomElement.zoom = value;
771     },false);
772     
773     var range = document.createElement('input');
774     range.setAttribute('min',min);
775     range.setAttribute('max',max);
776     range.setAttribute('step',precision);
777     range.setAttribute('value',value); 
778     range.setAttribute('type','range');
779     range.setAttribute('style','-webkit-appearance: slider-vertical; height: 100%; width: 1em; position: absolute; top: 0px; margin-top: 2em; left: 50%; margin-left: -0.5em;');
780 
781     if (range.type == 'range') {
782         
783         range.addEventListener('change',function() {
784             zoomElement.zoom = this.value;
785         },false);
786         zoomElement.registerEvent('zoomChange',function() {
787             range.value = zoomElement.zoom;
788         },false);
789         controls_container.appendChild(range);
790         
791     } else {
792         if (! zoomIn.addEventListener) {
793             var addevlis = function(name,func) {
794                 this.attachEvent(name,func);
795             };
796             zoomIn.addEventListener = addevlis;
797             reset.addEventListener = addevlis;
798             zoomOut.addEventListener = addevlis;        
799         }
800         zoomIn.addEventListener('click',function() {
801             zoomElement.zoom += precision;
802         },false);
803         zoomOut.addEventListener('click',function() {
804             zoomElement.zoom -= precision;
805         },false);
806 
807         controls_container.appendChild(zoomOut);
808         controls_container.appendChild(zoomIn);
809     }
810 
811     this._scrollZoomControls(controls_container,zoomElement,precision);
812 
813     return controls_container;
814 };

816 /**
817  * Connect the scroll wheel to the controls to control zoom
818  */
819 GOMap.Diagram._scrollZoomControls = function(controlElement,target,precision) {
820     var hookEvent = function(element, eventName, callback) {
821       if (typeof(element) == 'string') {
822         element = document.getElementById(element);
823       }
824 
825       if (element == null) {
826         return;
827       }
828 
829       if (element.addEventListener) {
830         if (eventName == 'mousewheel') {
831           element.addEventListener('DOMMouseScroll', callback, false);  
832         }
833         element.addEventListener(eventName, callback, false);
834       } else if (element.attachEvent) {
835         element.attachEvent("on" + eventName, callback);
836       }
837     };
838 
839     var mouseWheel = function(e) {
840       e = e ? e : window.event;
841       var wheelData = e.detail ? e.detail * -1 : e.wheelDelta;
842       if (wheelData > 0) {
843         target.zoom = target.zoom += precision;
844       } else {
845         target.zoom = target.zoom -= precision;
846       }
847       if (e.preventDefault) {
848         e.preventDefault();
849       }
850       return false;
851     };
852 
853     var isFF = false;
854 
855     if (navigator.userAgent.indexOf('Gecko') >= 0) {
856       isFF = parseFloat(navigator.userAgent.split('Firefox/')[1]) || undefined;
857     }                         
858 
859     if (isFF && svgweb.getHandlerType() == 'native') {
860       hookEvent(controlElement, 'mousewheel',
861                 mouseWheel);
862     } else {
863       hookEvent(controlElement, 'mousewheel', mouseWheel);
864     }
865 };
