  5 if ( typeof MASCP == 'undefined' || typeof MASCP.Service == 'undefined' ) {
  6     throw "MASCP.Service is not defined, required class";

 10 /** Default class constructor
 11  *  @class      Service class that will retrieve data from TAIR for a given AGI.
 12  *              Data is transferred using XML.
 13  *  @param      {String} agi            Agi to look up
 14  *  @param      {String} endpointURL    Endpoint URL for this service
 15  *  @extends    MASCP.Service
 16  */
 17 MASCP.TairReader = MASCP.buildService(function(data) {
 18                         this._data = data;
 19                         return this;
 20                     });
 21 
 22 MASCP.TairReader.prototype.requestData = function()
 23 {
 24     return {
 25         type: "POST",
 26         dataType: "xml",
 27         data: { 'name'  : this.agi,
 28                 'type'  : 'aa_sequence',
 29                 'service' : 'tair' 
 30         }
 31     };
 32 };
 33 
 34 
 35 MASCP.TairReader.Result.prototype.getSequence = function() {
 36 	var inputs = this._data.getElementsByTagName('input');
 37 	for (var i = 0; i < inputs.length; i++ ) {
 38 		if (inputs[i].getAttribute('name') == 'sequence') {
 39 			return inputs[i].getAttribute('value');
 40 		}
 41 	}
 42 	return null;
 43 };
